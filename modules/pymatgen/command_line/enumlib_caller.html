This module implements an interface to enumlib, Gus Hart's excellent Fortran
code for enumerating derivative structures. If you use this module, please cite
the following:

Gus L. W. Hart and Rodney Forcade, "A new approach to generating derivative
structures," Phys. Rev B. (2012).

This module depends on a compiled enumlib with the executables multi_enum.x and
make_str.x available in the path.
'''

[docs]    def gen_input_file(self, output_dir):
        """
        Generate the necessary struct_enum.in file for enumlib.
        """
        coord_format = "{:.6f} {:.6f} {:.6f}"
        struct = self._structure
        comp = struct.composition

        output = [struct.formula]
        output.append('bulk')
        for vec in self._structure.lattice.matrix:
            output.append(coord_format.format(*vec))
        output.append("{}".format(len(comp.elements)))
        output.append("{}".format(len(struct)))
        sorted_struct = struct.get_sorted_structure()
        i = 0
        all_species = sorted(struct.composition.elements, key=lambda el: el.X)
        for sp_and_occu, sites in itertools.groupby(sorted_struct,
                                        key=lambda site: site.species_and_occu):
            sp_label = []
            for sp in sp_and_occu.keys():
                sp_label.append(all_species.index(sp))
            sp_label = "/".join(["{}".format(i) for i in sorted(sp_label)])
            for site in sites:
                output.append(coord_format.format(*site.coords) + " " + sp_label)
        self.all_species = all_species
        output.append("{} {}".format(self._min_cell_size, self._max_cell_size))
        output.append("0.01")
        output.append("full")
        for sp in all_species:
            conc = comp.get_atomic_fraction(sp)
            if abs(conc * 100 - round(conc * 100)) < 1e-5:
                output.append("{} {} {}".format(int(round(conc * 100)),
                                                int(round(conc * 100)), 100))
            else:
                min_conc = int(math.floor(conc * 100))
                output.append("{} {} {}".format(min_conc - 1, min_conc + 1, 100))
        output.append("")
        temp_dir = tempfile.mkdtemp()
        with open(os.path.join(temp_dir, "struct_enum.in"), "w") as f:
            f.write("\n".join(output))
        os.chdir(temp_dir)
        p = subprocess.Popen(['/Users/shyuepingong/bin/multienum.x'],
                             stdout=subprocess.PIPE,
                             stdin=subprocess.PIPE, close_fds=True)
        output = p.communicate()[0]
        self.convert(temp_dir)
        shutil.rmtree(temp_dir)

    0.50000000     0.50000000     0.50000000     0/1   # d01 d-vector, labels: 0/1

    1 2        # Cell sizes to enumerate
    .0001      # Finite precision parameter
    full       # generate a full listing (e.g., consider AAAB!=BBBA)
    50 100 100
    50 100 100
    '''

        structs = []
        for n in range(1, 3):
            rs = subprocess.Popen(['/Users/shyuepingong/bin/makestr.x',
                                   'struct_enum.out', str(n)],
                                  stdout=subprocess.PIPE,

            rs.communicate()
            with open('vasp.{:06d}'.format(n)) as f:
                data = f.read()
                data = re.sub("scale factor", "1.0", data)
