[docs]def computed_entries_from_json(jsonstr):
    json_list = json.loads(jsonstr)
    entries = list()
    for entry in json_list:
        sid = entry['structureid']
        uncorrected_energy = entry['uncorr_energy']
        composition = Composition.from_formula(entry['formulasum'])
        correction = entry['correction']
        if 'structure' in entry:
            structure = Structure.from_dict(entry['structure'])
            entries.append(ComputedStructureEntry(sid, composition, uncorrected_energy, correction, structure))
        else:
            entries.append(ComputedEntry(sid, composition, uncorrected_energy, correction))
    return entries
