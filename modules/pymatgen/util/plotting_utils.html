    if not height:
        height = int(width * golden_ratio)
    plt.figure(figsize = (width, height), facecolor = 'w')
    plt.ylabel('', fontsize = 36)
    plt.xlabel('', fontsize = 36)
    plt.xticks(fontsize = 30)

    plt.title('', fontsize = 26)
